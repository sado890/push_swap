#include "push_swap.h"

void	run_strategy(t_node *a, t_node *b, t_strategy strat, t_info *bench)
{
	if (strat == LOW_DISORDER)
	{
		bench->notation = "O(n^2)";
		simple_sort(a, b, bench);
	}
	else if (strat == MEDIUM_DISORDER)
	{
		bench->notation = "O(n√n)";
		medium_sort(a, b, bench);
	}
	else if (strat == HIGH_DISORDER)
	{
		bench->notation = "O(n log n)";
		complex_sort(a, b, bench);
	}
	else if (strat == ADAPTIVE)
	{
		execute_adaptive_sort(a, b, bench);
	}
	else
		write(1, "error\n", 7);
}

void	execute_adaptive_sort(t_node *a, t_node *b, t_info *bench)
{
	float		disorder;
	t_strategy	strat;

	set_current_and_target_idx(a);
	disorder = compute_disorder(a);
	if (disorder < 0.2f)
		strat = LOW_DISORDER;
	else if (disorder >= 0.2f && disorder <= 0.5f)
		strat = MEDIUM_DISORDER;
	else
		strat = HIGH_DISORDER;
	run_strategy(a, b, strat, bench);
}

int	handle_small_size(t_node *a, t_node *b, int size, t_info *bench)
{
	if (size == 5)
		sort_5(a, b, bench);
	else if (size == 4)
		sort_4(a, b, bench);
	else if (size == 3)
		sort_3(a, bench);
	else if (size == 2)
		sort_2(a, bench);
	else
		return (-1);
	return (0);
}

static void	radix_sort(t_node **a, t_node **b, int size, t_info *bench)
{
	int	max_bits;
	int	i;
	int	j;

	max_bits = 0;
	while ((size >> max_bits) != 0)
		max_bits++;
	i = 0;
	while (i < max_bits)
	{
		j = 0;
		while (j < size)
		{
			if ((((*a)->target_idx >> i) & 1) == 0)
				pb(b, a, 1, bench);
			else
				ra(a, 1, bench);
			j++;
		}
		while (*b)
			pa(a, b, 1, bench);
		i++;
	}
}

void	complex_sort(t_node *a, t_node *b, t_info *bench)
{
	int	size;
	int	i;

	set_current_and_target_idx(a);
	size = (int)stack_len(a);
	i = handle_small_size(a, b, size, bench);
	if (i != -1)
		return ;
	radix_sort(&a, &b, size, bench);
}

static float	count_inversions(t_node *node)
{
	t_node	*temp;
	float	counter;

	temp = node->next;
	counter = 0;
	while (temp)
	{
		if (node->value > temp->value)
		{
			counter++;
		}
		temp = temp->next;
	}
	return (counter);
}

float	compute_disorder(t_node *a)
{
	float	mistakes;
	size_t	len;
	float	total_pairs;
	t_node	*temp;

	len = 0;
	total_pairs = 0;
	temp = a;
	mistakes = 0;
	while (temp)
	{
		mistakes += count_inversions(temp);
		temp = temp->next;
	}
	len = stack_len(a);
	if (len <= 1)
		return (0.0f);
	total_pairs = len * (len - 1) / 2;
	return (mistakes / total_pairs);
}

t_node	*sort_2(t_node *a, t_info *bench)
{
	if (a->target_idx > a->next->target_idx)
		sa(&a, 1, bench);
	return (a);
}

t_node	*sort_3(t_node *a, t_info *bench)
{
	set_current_and_target_idx(a);
	if (a->target_idx == 0 && a->next->target_idx == 2)
	{
		sa(&a, 1, bench);
		ra(&a, 1, bench);
	}
	else if (a->target_idx == 1 && a->next->target_idx == 0)
		sa(&a, 1, bench);
	else if (a->target_idx == 1 && a->next->target_idx == 2)
		rra(&a, 1, bench);
	else if (a->target_idx == 2 && a->next->target_idx == 0)
		ra(&a, 1, bench);
	else if (a->target_idx == 2 && a->next->target_idx == 1)
	{
		sa(&a, 1, bench);
		rra(&a, 1, bench);
	}
	set_current_and_target_idx(a);
	return (a);
}

t_node	*sort_4(t_node *a, t_node *b, t_info *bench)
{
	while (a->target_idx != 0)
		ra(&a, 1, bench);
	pb(&b, &a, 1, bench);
	a = sort_3(a, bench);
	pa(&a, &b, 1, bench);
	set_current_and_target_idx(a);
	return (a);
}

t_node	*sort_5(t_node *a, t_node *b, t_info *bench)
{
	while (a->target_idx != 0)
		ra(&a, 1, bench);
	pb(&b, &a, 1, bench);
	set_current_and_target_idx(a);
	sort_4(a, b, bench);
	pa(&a, &b, 1, bench);
	set_current_and_target_idx(a);
	return (a);
}

void	set_current_and_target_idx(t_node *stack)
{
	set_current_indices(stack);
	set_target_indices(stack);
}

void	set_current_indices(t_node *stack)
{
	int	index;

	index = 0;
	while (stack)
	{
		stack->curr_idx = index++;
		stack = stack->next;
	}
}

void	set_target_indices(t_node *stack)
{
	t_node	*temp;
	t_node	*current;
	int		count;

	current = stack;
	while (current)
	{
		count = 0;
		temp = stack;
		while (temp)
		{
			if (temp->value < current->value)
				count++;
			temp = temp->next;
		}
		current->target_idx = count;
		current = current->next;
	}
}

int	fill_stack(t_node **a, char **split)
{
	int	i;
	int	val;
	int	err;

	i = 0;
	while (split[i])
	{
		err = 0;
		val = ft_atoi_check(split[i], &err);
		if (err || is_duplicate(*a, val))
			return (-1);
		add_back(a, new_node(val));
		i++;
	}
	return (0);
}

static void	print_strategy(t_info *bench)
{

#include "push_swap.h"

static char	parse_flag(char *arg)
{
	if (arg[0] == '-' && arg[1] == '-')
	{
		if (ft_strncmp(arg, "--simple", 9) == 0)
			return ('S');
		else if (ft_strncmp(arg, "--medium", 9) == 0)
			return ('M');
		else if (ft_strncmp(arg, "--complex", 10) == 0)
			return ('C');
		else if (ft_strncmp(arg, "--adaptive", 11) == 0)
			return ('A');
		else if (ft_strncmp(arg, "--bench", 8) == 0)
			return ('B');
		else
			return ('E');
	}
	return ('N');
}

static int	parse_and_fill_all(t_node **a, int argc, char **argv, int i)
{
	char	**split;

	while (i < argc)
	{
		split = ft_split(argv[i]);
		if (!split || !split[0] || fill_stack(a, split) == -1)
		{
			if (split)
				free_split(split);
			return (-1);
		}
		free_split(split);
		i++;
	}
	return (0);
}

static void	set_strategy(char format, t_strategy *strat, t_info *bench)
{
	if (format == 'A' || format == 'N')
	{
		bench->strategy = ADAPTIVE;
		*strat = ADAPTIVE;
	}
	else if (format == 'S')
	{
		bench->strategy = LOW_DISORDER;
		*strat = LOW_DISORDER;
	}
	else if (format == 'M')
	{
		bench->strategy = MEDIUM_DISORDER;
		*strat = MEDIUM_DISORDER;
	}
	else if (format == 'C')
	{
		bench->strategy = HIGH_DISORDER;
		*strat = HIGH_DISORDER;
	}
}

int	main(int argc, char **argv)
{
	t_node		*a;
	t_node		*b;
	t_info		*bench;
	t_strategy	strat;
	char		format;
	int			i;

	if (argc < 2)
		return (0);
	a = NULL;
	b = NULL;
	i = 1;
	format = parse_flag(argv[1]);
	bench = malloc(sizeof(t_info));
	if (!bench)
		return (1);
	strat = ADAPTIVE;
	ft_memset(bench, 0, sizeof(t_info));
	if (format == 'B')
	{
		bench->bench_mode = 1;
		i = 2;
		format = 'A';
	}
	else if (format != 'N' && format != 'E')
		i++;
	if (parse_and_fill_all(&a, argc, argv, i) == -1)
	{
		free_stack(&a);
		free(bench);
		write(2, "Error\n", 6);
		return (1);
	}
	if (format == 'E')
	{
		free_stack(&a);
		free(bench);
		return (write(2, "Error\n", 6), 1);
	}
	set_current_and_target_idx(a);
	bench->disorder = compute_disorder(a);
	set_strategy(format, &strat, bench);
	run_strategy(a, b, strat, bench);
	if (bench->bench_mode == 1)
		print_bench(bench);
	free(bench);
	free_stack(&a);
	return (0);
}

#include "push_swap.h"

static int	get_cost(int size, int idx)
{
	if (idx <= size / 2)
		return (idx);
	return (idx - size);
}

static int	get_target_pos(t_node *a, int b_val)
{
	t_node	*tmp;
	int		best_idx;
	int		best_val;

	tmp = a;
	best_idx = -1;
	best_val = 2147483647;
	while (tmp)
	{
		if (tmp->value > b_val && tmp->value < best_val)
		{
			best_val = tmp->value;
			best_idx = tmp->curr_idx;
		}
		tmp = tmp->next;
	}
	if (best_idx != -1)
		return (best_idx);
	tmp = a;
	best_val = 2147483647;
	while (tmp)
	{
		if (tmp->value < best_val)
		{
			best_val = tmp->value;
			best_idx = tmp->curr_idx;
		}
		tmp = tmp->next;
	}
	return (best_idx);
}

static void	do_rotate(t_node **a, t_node **b, int ca, int cb, t_info *bench)
{
	while (ca > 0 && cb > 0)
	{
		rr(a, b, bench);
		ca--;
		cb--;
	}
	while (ca < 0 && cb < 0)
	{
		rrr(a, b, bench);
		ca++;
		cb++;
	}
	while (ca > 0)
	{
		ra(a, 1, bench);
		ca--;
	}
	while (ca < 0)
	{
		rra(a, 1, bench);
		ca++;
	}
	while (cb > 0)
	{
		rb(b, 1, bench);
		cb--;
	}
	while (cb < 0)
	{
		rrb(b, 1, bench);
		cb++;
	}
}

static void	push_back_cheapest(t_node **a, t_node **b, t_info *bench)
{
	t_node	*tmp;
	int		best;
	int		best_a;
	int		best_b;
	int		target;
	int		ca;
	int		cb;
	int		total;

	best = 2147483647;
	best_a = 0;
	best_b = 0;
	set_current_and_target_idx(*a);
	set_current_indices(*b);
	tmp = *b;
	while (tmp)
	{
		target = get_target_pos(*a, tmp->value);
		ca = get_cost(stack_len(*a), target);
		cb = get_cost(stack_len(*b), tmp->curr_idx);
		total = abs(ca) + abs(cb);
		if (total < best)
		{
			best = total;
			best_a = ca;
			best_b = cb;
		}
		tmp = tmp->next;
	}
	do_rotate(a, b, best_a, best_b, bench);
	pa(a, b, 1, bench);
}

static void	push_chunks(t_node **a, t_node **b, t_info *bench)
{
	int	chunk;
	int	pushed;

	chunk = 30;
	pushed = 0;
	while (*a)
	{
		set_current_and_target_idx(*a);
		if ((*a)->target_idx <= pushed)
		{
			pb(b, a, 1, bench);
			rb(b, 1, bench);
			pushed++;
		}
		else if ((*a)->target_idx <= pushed + chunk)
		{
			pb(b, a, 1, bench);
			pushed++;
		}
		else
			ra(a, 1, bench);
	}
}

static void	final_align(t_node **a, t_info *bench)
{
	int		min_idx;
	t_node	*tmp;

	min_idx = 0;
	set_current_and_target_idx(*a);
	tmp = *a;
	while (tmp)
	{
		if (tmp->target_idx == 0)
			break ;
		min_idx++;
		tmp = tmp->next;
	}
	if (min_idx <= (int)stack_len(*a) / 2)
	{
		while (min_idx--)
			ra(a, 1, bench);
	}
	else
	{
		while (min_idx++ < (int)stack_len(*a))
			rra(a, 1, bench);
	}
}

void	medium_sort(t_node *a, t_node *b, t_info *bench)
{
	t_node	*stack_a;
	t_node	*stack_b;
	int len;

	stack_a = a;
	stack_b = b;
	len = (int)stack_len(stack_a);
	if (handle_small_size(a, b, len, bench) != -1)
		return ;
	push_chunks(&stack_a, &stack_b, bench);
	while (stack_b)
		push_back_cheapest(&stack_a, &stack_b, bench);
	final_align(&stack_a, bench);
}

void	*ft_memset(void *s, int c, size_t n)
{
	unsigned char	*ptr;

	ptr = (unsigned char *)s;
	while (n--)
		*ptr++ = (unsigned char)c;
	return (s);
}

t_node	*new_node(int value)
{
	t_node	*node;

	node = malloc(sizeof(t_node));
	if (!node)
		return (NULL);
	node->value = value;
	node->curr_idx = -1;
	node->target_idx = -1;
	node->next = NULL;
	return (node);
}

void	add_back(t_node **node, t_node *new)
{
	t_node	*last;

	if (!node || !new)
		return ;
	if (!*node)
	{
		*node = new;
		return ;
	}
	last = *node;
	while (last->next)
		last = last->next;
	last->next = new;
}

size_t	stack_len(t_node *node)
{
	size_t	len;

	len = 0;
	while (node)
	{
		len++;
		node = node->next;
	}
	return (len);
}

int	is_clear(char *str)
{
	int	i;

	i = 0;
	if (!str || !str[0])
		return (0);
	if (str[i] == '-' || str[i] == '+')
		i++;
	if (!str[i])
		return (0);
	while (str[i])
	{
		if (!(str[i] >= '0' && str[i] <= '9'))
			return (0);
		i++;
	}
	return (1);
}

int	is_duplicate(t_node *stack, int val)
{
	while (stack)
	{
		if (stack->value == val)
			return (1);
		stack = stack->next;
	}
	return (0);
}

int	ft_atoi_check(char *str, int *error)
{
	long	res;
	int		sign;
	int		i;

	res = 0;
	sign = 1;
	i = 0;
	if (!is_clear(str))
		return (*error = 1, 0);
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] == '-' || str[i] == '+')
		if (str[i++] == '-')
			sign = -1;
	while (str[i] >= '0' && str[i] <= '9')
	{
		res = res * 10 + (str[i++] - '0');
		if ((res * sign) > 2147483647 || (res * sign) < -2147483648)
			return (*error = 1, 0);
	}
	return ((int)(res * sign));
}

static int	get_min_idx(t_node *a)
{
	t_node	*temp;

	temp = a;
	while (temp)
	{
		if (temp->target_idx == 0)
			return (temp->curr_idx);
		temp = temp->next;
	}
	return (0);
}

static void	rotate_to_top(t_node **a, int min_idx, int size, t_info *bench)
{
	if (min_idx <= size / 2)
	{
		while (min_idx-- > 0)
			ra(a, 1, bench);
	}
	else
	{
		while (min_idx++ < size)
			rra(a, 1, bench);
	}
}

void	simple_sort(t_node *a, t_node *b, t_info *bench)
{
	int		size;
	int		min_idx;

	size = (int)stack_len(a);
	while ((int)stack_len(a) > 3)
	{
		set_current_and_target_idx(a);
		size = (int)stack_len(a);
		min_idx = get_min_idx(a);
		rotate_to_top(&a, min_idx, size, bench);
		pb(&b, &a, 1, bench);
	}
	a = sort_3(a, bench);
	while (b)
		pa(&a, &b, 1, bench);
}

static void	fill_split(char *str, char **res)
{
	int	i;
	int	k;
	int	j;
	int	l;

	i = 0;
	k = 0;
	while (str[i])
	{
		while (str[i] == ' ')
			i++;
		if (str[i])
		{
			j = i;
			while (str[j] && str[j] != ' ')
				j++;
			res[k] = malloc(j - i + 1);
			l = 0;
			while (i < j)
				res[k][l++] = str[i++];
			res[k][l] = '\0';
			k++;
		}
	}
	res[k] = NULL;
}

char	**ft_split(char *str)
{
	char	**res;

	res = malloc(sizeof(char *) * (ft_strlen(str) + 1));
	if (!res)
		return (NULL);
	fill_split(str, res);
	return (res);
}

int	ft_strncmp(const char *s1, const char *s2, size_t n)
{
	size_t	i;

	i = 0;
	while (i < n && (s1[i] || s2[i]))
	{
		if ((unsigned char)s1[i] != (unsigned char)s2[i])
			return ((unsigned char)s1[i] - (unsigned char)s2[i]);
		i++;
	}
	return (0);
}
